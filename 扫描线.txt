void solve() {
    int n;
    cin >> n;
  
    vector<int> a(n);  
    vector<int> vals; 
    
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        vals.push_back(a[i]); 
    }
    
    
    sort(vals.begin(), vals.end());
    vals.erase(unique(vals.begin(), vals.end()), vals.end());
    
    for (int i = 0; i < n; i++) {
        int id = lower_bound(vals.begin(), vals.end(), a[i]) - vals.begin() + 1;
        cout << a[i] << "  " << id << "\n";
    }
}
