int qpow(int a, int b) {
    int res = 1;
    a %= MOD;
    while (b > 0) {
        if (b & 1) res = res * a % MOD;
        a = a * a % MOD;
        b >>= 1;
    }
    return res;
}
//直接递推组合数
const int MOD = 1e9 + 7; // 任意正整数均可
const int MAXN = 2005;
long long C[MAXN][MAXN];

void init_C() {
    for (int i = 0; i < MAXN; i++) {
        C[i][0] = 1; // 从 i 个物品里选 0 个，方案数为 1
        for (int j = 1; j <= i; j++) {
            C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % MOD;
        }
    }
}

// 预处理阶乘与逆元，复杂度 O(N)
void init() {
    fact[0] = 1;
    inv[0] = 1;
    for (int i = 1; i < MAXN; i++) {
        fact[i] = fact[i - 1] * i % MOD;
    }
    // 先用快速幂求出最大项的逆元
    inv[MAXN - 1] = qpow(fact[MAXN - 1], MOD - 2);
    // 倒推求其他项的逆元：(n-1)! 的逆元 = n! 的逆元 * n
    for (int i = MAXN - 2; i >= 1; i--) {
        inv[i] = inv[i + 1] * (i + 1) % MOD;
    }
}

// O(1) 查询组合数 C(n, m)
long long C(int n, int m) {
    if (m < 0 || m > n) return 0;
    return fact[n] * inv[m] % MOD * inv[n - m] % MOD;
}

// O(1) 查询排列数 A(n, m)
long long A(int n, int m) {
    if (m < 0 || m > n) return 0;
    return fact[n] * inv[n - m] % MOD;
}

//数据范围大，且分母与模数不互质
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e6 + 5;
int primes[MAXN], cnt;
bool st[MAXN];

// 线性筛求素数
void get_primes(int n) {
    for (int i = 2; i <= n; i++) {
        if (!st[i]) primes[cnt++] = i;
        for (int j = 0; primes[j] <= n / i; j++) {
            st[i * primes[j]] = true;
            if (i % primes[j] == 0) break;
        }
    }
}

// 计算 n! 中包含素数 p 的个数
int get_p_cnt(int n, int p) {
    int res = 0;
    while (n) {
        res += n / p;
        n /= p;
    }
    return res;
}

// 快速幂
long long qpow(long long a, long long b, long long mod) {
    long long res = 1;
    a %= mod;
    while (b > 0) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

// 计算 C(n, m) % mod，mod 可以是任意数字
long long C_arbitrary_mod(int n, int m, long long mod) {
    get_primes(n);
    long long res = 1;
    for (int i = 0; i < cnt; i++) {
        int p = primes[i];
        // 分子包含的 p 的个数 减去 分母包含的 p 的个数
        int k = get_p_cnt(n, p) - get_p_cnt(m, p) - get_p_cnt(n - m, p);
        res = res * qpow(p, k, mod) % mod;
    }
    return res;
}
