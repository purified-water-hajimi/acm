// 返回 a 和 b 的最大公约数，同时通过引用传出对应的 x 和 y
int exgcd(int a, int b, int &x, int &y) {
    if (b == 0) {
        x = 1;
        y = 0;
        return a; // 当 b 为 0 时，最大公约数就是 a
    }
    
    int x1, y1;
    // 递归调用，先求出下一层的 x1 和 y1
    int d = exgcd(b, a % b, x1, y1);
    
    // 根据下一层的 x1, y1 推导当前层的 x, y
    x = y1;
    y = x1 - (a / b) * y1;
    
    return d; // 返回最大公约数
}
