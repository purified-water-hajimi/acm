int l = 0, r = 0; // 左右指针都在起点
int ans = 0;    
while (r < n) {
    // 1. 右指针主动把新元素拉进窗口，更新窗口状态
    add(a[r]);
    // 2. 检查窗口状态是否【非法】（或者是否【满足收缩条件】）
    // 注意：这里必须是 while，因为左指针可能需要连续吐出好几个元素才能恢复合法
    while (l <= r && check_is_invalid()) {
        // 左指针把元素踢出窗口，更新窗口状态
        l++; // 左指针向右收缩
    }
    
    // 3. 经过上面的 while，此时的窗口 [l, r] 一定是合法的
    // 在这里更新最终答案
    ans = max(ans, r - l + 1); 
    
    // 4. 右指针继续去拉下一个元素
    r++;
}
