int l = 0, r = n - 1; // 1. 初始化左右边界
int ans = 0;

while (l < r) {       // 2. 核心条件：左右指针未相遇 (根据题意有时是 l <= r)
    if (满足完美条件(a[l], a[r])) {
        // 记录答案
        l++; 
        r--; // 找到后可能两个指针都要收缩
    } 
    else if (当前状态比目标小(a[l], a[r])) {
        l++; // 只能让小的变大，左指针右移
    } 
    else {
        r--; // 只能让大的变小，右指针左移
    }
}
